*&---------------------------------------------------------------------*
*&      Form FETCH_DATA
*&---------------------------------------------------------------------*
*&      Get data into internal table
*&---------------------------------------------------------------------*
FORM fetch_data.
  REFRESH gt_sflight.

  SELECT carrid
         connid
         fldate
         zdesc
    INTO TABLE gt_sflight
    FROM zsflight
    WHERE carrid IN so_car
      AND fldate IN so_dat.

  IF sy-subrc <> 0.
    CLEAR gt_sflight.
  ENDIF.
ENDFORM.

*&---------------------------------------------------------------------*
*&      Form SAVE_RECORD
*&---------------------------------------------------------------------*
*&      Simpan perubahan ke database
*&
*&      PENTING: SFLIGHT adalah tabel standar SAP.
*&      Di sistem production gunakan Z-table sendiri.
*&
*&      iv_mode: 'C' = Create, 'U' = Update, 'D' = Delete
*&---------------------------------------------------------------------*
FORM save_record USING iv_data TYPE ty_sflight
                       iv_mode TYPE char1.
  CASE iv_mode.

    WHEN 'C'.
      DATA ls_insert TYPE zsflight.
      MOVE-CORRESPONDING iv_data TO ls_insert.
      INSERT zsflight FROM ls_insert.
      IF sy-subrc <> 0.
        ROLLBACK WORK.
        MESSAGE |Gagal menyimpan record baru| TYPE 'E'.
        RETURN.
      ENDIF.

    WHEN 'U'.
      UPDATE zsflight
        SET zdesc    = iv_data-zdesc
        WHERE carrid = iv_data-carrid
          AND connid = iv_data-connid
          AND fldate = iv_data-fldate.
      IF sy-subrc <> 0.
        ROLLBACK WORK.
        MESSAGE |Gagal mengupdate record| TYPE 'E'.
        RETURN.
      ENDIF.

    WHEN 'D'.
      DELETE FROM zsflight
        WHERE carrid = iv_data-carrid
          AND connid = iv_data-connid
          AND fldate = iv_data-fldate.
      IF sy-subrc <> 0.
        ROLLBACK WORK.
        MESSAGE |Gagal menghapus record| TYPE 'E'.
        RETURN.
      ENDIF.

  ENDCASE.

  COMMIT WORK AND WAIT.
ENDFORM.
