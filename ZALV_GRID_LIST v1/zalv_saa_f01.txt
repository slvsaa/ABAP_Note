FORM fetch_data.
  REFRESH gt_sflight.

  SELECT carrid
         connid
         fldate
         price
         currency
         planetype
         seatsmax
         seatsocc
         paymentsum
    INTO TABLE gt_sflight
    FROM sflight
    WHERE carrid IN so_car
      AND fldate IN so_dat.

  IF sy-subrc <> 0.
    CLEAR gt_sflight.
  ENDIF.
ENDFORM.
