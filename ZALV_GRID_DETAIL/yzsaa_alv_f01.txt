FORM fetch_data.
  REFRESH gt_po.

  SELECT ebeln    "Nomor PO
         lifnr    "Nomor Vendor
         bedat    "Tanggal PO
    INTO TABLE gt_po
    FROM ekko
    WHERE ebeln IN so_ebn .

  IF sy-subrc <> 0.
    CLEAR gt_po.
  ENDIF.
ENDFORM.
